/* Private define ------------------------------------------------------------*/
/* USER CODE BEGIN PD */
#define NUM_SENSORS 4
/* USER CODE END PD */

/* Private variables ---------------------------------------------------------*/
ADC_HandleTypeDef hadc1;
DMA_HandleTypeDef hdma_adc1;

TIM_HandleTypeDef htim1;
TIM_HandleTypeDef htim3;
TIM_HandleTypeDef htim4;

/* USER CODE BEGIN PV */
// ADC DMA Raw Buffer
uint16_t adc_buffer[NUM_SENSORS];

// Linearized Distance Variables (in cm)
float distance_left = 0.0f;
float distance_front_left = 0.0f;
float distance_front_right = 0.0f;
float distance_right = 0.0f;

// Encoder Variables
int16_t encoder_left_ticks = 0;
int16_t encoder_right_ticks = 0;
/* USER CODE END PV */

void set_motor_right(int32_t speed); // Changed from int16_t
/* USER CODE END PFP */

/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* MCU Configuration--------------------------------------------------------*/
  HAL_Init();

  /* Configure the system clock */
  SystemClock_Config();

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_DMA_Init();
  MX_ADC1_Init();
  MX_TIM3_Init();
  MX_TIM4_Init();
  MX_TIM1_Init();

  /* USER CODE BEGIN 2 */
  // 1. Enable the TB6612FNG Driver out of Standby Mode
  HAL_GPIO_WritePin(GPIOB, GPIO_PIN_3, GPIO_PIN_SET);

  // 2. Start PWM Signals for Motor Speed Control
  HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_1); // PWMA (Left Motor)
  HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_4); // PWMB (Right Motor)

  // 3. Start Hardware Quadrature Encoder Timers
  HAL_TIM_Encoder_Start(&htim3, TIM_CHANNEL_ALL); // Left Encoder
  HAL_TIM_Encoder_Start(&htim4, TIM_CHANNEL_ALL); // Right Encoder

  // 4. Start Continuous Background ADC Streaming via DMA
  HAL_ADC_Start_DMA(&hadc1, (uint32_t *)adc_buffer, NUM_SENSORS);
  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
       /* --- STEP 1: READ DISTANCE SENSORS (Background Tracking) --- */
       distance_left        = convert_adc_to_cm(adc_buffer[0]);
       distance_front_left  = convert_adc_to_cm(adc_buffer[1]);
       distance_front_right = convert_adc_to_cm(adc_buffer[2]);
       distance_right       = convert_adc_to_cm(adc_buffer[3]);

       /* --- STEP 2: READ ENCODER POSITIONS --- */
       encoder_left_ticks  = (int16_t)__HAL_TIM_GET_COUNTER(&htim3);
       encoder_right_ticks = (int16_t)__HAL_TIM_GET_COUNTER(&htim4);

       /* --- STEP 3: ALTERNATING TIMED MOTION PROFILE --- */

              // 1. DRIVE FORWARD AT ABSOLxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxUTE MAX SPEED FOR 5 SECONDS
              set_motor_left(65535);  // CHAMaximumNGED: Maximum timer period for 100% Duty Cycle
              set_motor_right(65535); // CHANGED:  timer period for 100% Duty Cycle
              HAL_Delay(5000);

              // 2. ACTIVE SOLID BRAKE COOLDOWN
              set_motor_left(0);
              set_motor_right(0);
              HAL_Delay(200);

              // 3. DRIVE BACKWARD (Kept safe at -30000, or change to -65535 if using updated functions below)
              set_motor_left(-30000);
              set_motor_right(-30000);
              HAL_Delay(5000);

       // 4. ACTIVE SOLID BRAKE COOLDOWN
       set_motor_left(0);
       set_motor_right(0);
       HAL_Delay(200);

    /* USER CODE END WHILE */
  }
}

/**
  * @brief  Linearizes the non-linear analog Sharp IR output curve to real centimeters.
  */
float convert_adc_to_cm(uint16_t adc_val)
{
	if (adc_val < 150) return 15.0f;

    float voltage = (float)adc_val * (3.3f / 4095.0f);
    float distance = 5.3f * powf(voltage, -1.15f);

    if (distance > 15.0f) return 15.0f;
    if (distance < 2.0f)  return 2.0f;

    return distance;
}

/**
  * @brief  Controls Left Motor Direction and Speed via TB6612FNG Channel A
  * @note   Changed parameter type to int32_t to support up to 65535 PWM period.
  */
void set_motor_left(int32_t speed)
{
    if (speed > 0)
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_10, GPIO_PIN_RESET);
        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, speed);
    }
    else if (speed < 0)
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_10, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, -speed);
    }
    else
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_10, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, 0);
    }
}

/**
  * @brief  Controls Right Motor Direction and Speed via TB6612FNG Channel B
  * @note   Changed parameter type to int32_t to support up to 65535 PWM period.
  */
void set_motor_right(int32_t speed)
{
    if (speed > 0)
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_12, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_15, GPIO_PIN_RESET);
        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_4, speed);
    }
    else if (speed < 0)
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_12, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_15, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_4, -speed);
    }
    else
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_12, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_15, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_4, 0);
    }
}
