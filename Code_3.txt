/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{
  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_DMA_Init();
  MX_ADC1_Init();
  MX_TIM3_Init();
  MX_TIM4_Init();
  MX_TIM1_Init();

  /* USER CODE BEGIN 2 */
  // 1. Enable the TB6612FNG Driver out of Standby Mode
  HAL_GPIO_WritePin(GPIOB, GPIO_PIN_3, GPIO_PIN_SET);

  // 2. Start PWM Signals for Motor Speed Control
  HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_1); // PWMA (Left Motor)
  HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_4); // PWMB (Right Motor)

  // 3. Start Hardware Quadrature Encoder Timers
  HAL_TIM_Encoder_Start(&htim3, TIM_CHANNEL_ALL); // Left Encoder
  HAL_TIM_Encoder_Start(&htim4, TIM_CHANNEL_ALL); // Right Encoder

  // 4. Start Continuous Background ADC Streaming via DMA
  HAL_ADC_Start_DMA(&hadc1, (uint32_t *)adc_buffer, NUM_SENSORS);
  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while(1)
  {
      /* --- 1. READ AND LINEARIZE SENSORS (2-15cm Model) --- */
      distance_left        = convert_adc_to_cm(adc_buffer[0]);
      distance_front_left  = convert_adc_to_cm(adc_buffer[1]);
      distance_front_right = convert_adc_to_cm(adc_buffer[2]);
      distance_right       = convert_adc_to_cm(adc_buffer[3]);

      /* --- 2. UPDATE WALL BOOLEAN FLAGS --- */
      wall_left  = (distance_left < WALL_THRESHOLD);
      wall_right = (distance_right < WALL_THRESHOLD);

      // Front wall triggers if either front sensor gets too close
      wall_front = ((distance_front_left < FRONT_THRESHOLD) ||
                    (distance_front_right < FRONT_THRESHOLD));

      /* --- 3. NAVIGATION STATE LOGIC --- */
      if (wall_front)
      {
          // We found a front wall! Stop immediately.
          stop_robot();
          HAL_Delay(200);

          if (!wall_left)
          {
              // Left is open -> Turn Left
              set_motor_left(-2000);
              set_motor_right(2000);
              HAL_Delay(330);
          }
          else if (!wall_right)
          {
              // Right is open -> Turn Right
              set_motor_left(2000);
              set_motor_right(-2000);
              HAL_Delay(330);
          }
          else
          {
              // Dead end -> 180 Degree Turn around
              set_motor_left(2000);
              set_motor_right(-2000);
              HAL_Delay(650);
          }

          // Stop cleanly post-turn to stabilize DMA readings
          stop_robot();
          HAL_Delay(200);
      }
      else
      {
          /* --- 4. WALL FOLLOWING / STRAIGHT PROPORTIONAL CONTROL --- */
          float error = 0.0f;

          // If inside a corridor (both walls tracked), stay dead center
          if (wall_left && wall_right)
          {
              error = distance_left - distance_right;
          }
          // If only tracking the left wall
          else if (wall_left)
          {
              error = distance_left - 6.0f; // Try to stay 6cm away from left wall
          }
          // If only tracking the right wall
          else if (wall_right)
          {
              error = 6.0f - distance_right; // Try to stay 6cm away from right wall
          }
          else
          {
              // No side walls seen! Drive completely straight
              error = 0.0f;
          }

          // Calculate Steering Adjustments
          float correction = KP_WALL * error;

          int32_t left_speed  = BASE_SPEED - (int32_t)correction;
          int32_t right_speed = BASE_SPEED + (int32_t)correction;

          // Enforce PWM Bounds (Hardware Ceiling: 4199)
          if (left_speed > 3500)  left_speed = 3500;
          if (right_speed > 3500) right_speed = 3500;
          if (left_speed < 800)   left_speed = 800;
          if (right_speed < 800)  right_speed = 800;

          set_motor_left(left_speed);
          set_motor_right(right_speed);
      }

      HAL_Delay(15); // Loop pacing
      /* USER CODE END WHILE */

      /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}

/**
  * @brief  Converts Sharp IR raw ADC value to distance in centimeters (2cm to 15cm Model)
  */
float convert_adc_to_cm(uint16_t adc_val)
{
    // Avoid dividing by zero or processing dead/disconnected sensor values
    if (adc_val < 150) return 15.0f;

    // Transform 12-bit raw integer to analog voltage scalar (3.3V baseline)
    float voltage = ((float)adc_val * 3.3f) / 4095.0f;

    // Custom inverse linearization curve mapped directly to 2-15cm Sharp data
    float distance = 5.3f / (voltage - 0.1f);

    // Hard boundary clamps to keep controller stable
    if (distance < 2.0f)  return 2.0f;
    if (distance > 15.0f) return 15.0f;

    return distance;
}

/**
  * @brief  Sets speed and direction for the Left Motor
  */
void set_motor_left(int32_t speed)
{
    if (speed >= 0)
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_10, GPIO_PIN_RESET);
    }
    else
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_10, GPIO_PIN_SET);
        speed = -speed;
    }
    __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, speed);
}

/**
  * @brief  Sets speed and direction for the Right Motor
  */
void set_motor_right(int32_t speed)
{
    if (speed >= 0)
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_12, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_15, GPIO_PIN_RESET);
    }
    else
    {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_12, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_15, GPIO_PIN_SET);
        speed = -speed;
    }
    __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_4, speed);
}
